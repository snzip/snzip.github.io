   ```bash

   git push -u origin main

   # 获取图片链接：
   # https://raw.githubusercontent.com/yourusername/ai-images/main/chart1-employment-forecast.png

   # https://cdn.jsdelivr.net/gh/yourusername/ai-images@main/chart1-employment-forecast.png
   ```

